Консоль - интерфейс для работы с ОС

#НАВИГАЦИЯ
cd - перемещение по каталогу
cd Desktop/JS - перемещение в JS
cd .. - на одну папку назад
cd ../.. - на две
cd ~ - переход в домашнюю директорию
cd / - переход в корневую папку 
ls - просмотр списка файлов в текущем каталоге
ls -l -a
pwd - напечатать к терминал текущий каталог - /home/Desctop/JS

#СОЗДАНИЕ ФАЙЛОВ и ПАПОК
mkdir folder_name - создать папку folder_name
mkdir one two - создать две папки - one и two
mkdir -p three/inner-folder - создать вложенные папки
//-p говорит что нужно создать все папки по пути к последней
mkdir -p {css/layout,html/pages} - создать несколько папок с вложениями

touch filename.txt - создать файлов
ls -la > testLA.txt - создать фалй и поместить в него вывод комманды ls
echo > echo.txt - создать файл и напечатать в него пустоту

#УДАЛНИЕ ФАЙЛОВ И ПАПОК
rm filename.txt - удалить файл (можно сразу несколько)
rmdir folder_name - удалить пустую папку
rm -r folder_name - удалить не пустую папку

#ПЕРЕМЕЩЕНИЕ И КОПИРОВАНИЕ
mv index.html test_folder - переместить файл в папку
mv index.html another.html - переместить и переименовать
mv -r test two - перемещение вложенных папок
cp new.html test - копировать файл в папку
cp -r test two - копирование вложенных папок (копировтаь test в two)
cp или mv или rm *.html - копировать/переместить все html файлы

#ИСПРАВЛЕНИЕ ОПЕЧАТОК
стрелка вверх - пролистывание последних комманды
ctrl + A - сместиться в начало комманды
ctrl + E - в конец
ctrl + U - очистка строки
^что исправить^на что - исправление последней комманды
//mkdor test - напечатали неправильно
//^mkdor^mkdir - исправление

#ПОИСК
find . -iname "test" - поиск в текущей папке по имени -i - не считать регистр
find . -size +15c - больше 15ти байт
find --help - документация к командам
